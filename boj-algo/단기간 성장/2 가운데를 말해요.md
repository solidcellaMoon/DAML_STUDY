```python
from bisect import bisect_left

n = int(input())

answer = []

#첫 번째 숫자 입력

for _ in range(n-1):
    num = int(input())
    idx = bisect_left(inputs,num)
    inputs.insert(idx,num)
    
    lens += 1
    
    if lens % 2 == 0:
        answer.append(inputs[lens//2 - 1])
        #print(inputs[len(inputs)//2 - 1])
    else:
```

## 이진탐색


```python
n = int(input())

#결과
answers = []

#말하는 번호 순서 저장
inputs = []

answer = []

for _ in range(n):
    inputs.append(int(input()))
    print(inputs)
    
    inputs.sort()
    
    if len(inputs) % 2 == 1:
        target = len(inputs)//2
    else:
        target = len(inputs)//2 - 1
        
    left, right = 0, n - 1

    while left <= right:

        mid = (left + right) // 2

        if mid == target:
            answer.append(inputs[mid])
            break

        elif mid < target:
            left = mid + 1

        else:
            right = mid - 1
        
answer
```

배열은 늘 정렬된다고 하자. [0]을 1번째라고 칠 때,

- 배열이 홀수개 => n//2 + 1
- 배열이 짝수개 => n//2 - 1

0 1 2 3 4 5 => 2
* len(n)//2  - 1

0 1 2 3 4 5 6 => 3
* len(n)//2

```python
import heapq as hp

#말하는 번호 순서
inputs = [1, 5, 2, 10, -99, 7, 5]

#힙큐
heaps = []

#결과
answers = []

def heapsort(arr):
    heap = []
    result = []
    
    for value in arr:
        hp.heappush(heap, value)
        
    for i in range(len(heap)):
        result.append(hp.heappop(heap))
        
    return result

for num in inputs:
    answers.append(num)
    answers = heapsort(answers)
    
    if len(answers) % 2 == 1:
        print(answers[len(answers)//2])

        print(answers[(len(answers)//2) - 1])
```
